Create `lib/uploads/replace.ts`:
```ts
"use server";
import { auth } from "@/lib/auth";
import { db } from "@/db/client";
import { document, documentVersion } from "@/db/schema";
import { eq, and } from "drizzle-orm";
import { validatePdf } from "./pdf-validate";
import { computeSha256 } from "./hash";
import { saveUploadedFile } from "./storage";
import { headers } from "next/headers";
import crypto from "crypto";
import path from "path";
import fs from "fs/promises";

export async function replaceDocumentPdfAction(
  documentId: string,
  formData: FormData
): Promise<{ ok: true; newVersion: number } | { ok: false; error: string }> {
  const session = await auth.api.getSession({ headers: await headers() });
  if (!session) return { ok: false, error: "unauthenticated" };

  const [doc] = await db.select().from(document).where(and(eq(document.id, documentId), eq(document.userId, session.user.id))).limit(1);
  if (!doc) return { ok: false, error: "not_found" };

  const file = formData.get("file");
  if (!(file instanceof File)) return { ok: false, error: "no_file" };

  const buffer = Buffer.from(await file.arrayBuffer());
  const validation = await validatePdf(buffer);
  if (!validation.ok) return { ok: false, error: validation.reason };

  const sha256 = computeSha256(buffer);
  const newVersionNumber = (doc.version ?? 1) + 1;
  // Save new file as {docId}-v{N}.pdf in data/uploads
  const newPath = path.join("data/uploads", `${documentId}-v${newVersionNumber}.pdf`);
  const absPath = path.resolve(process.cwd(), newPath);
  await fs.mkdir(path.dirname(absPath), { recursive: true });
  await fs.writeFile(absPath, buffer);

  // Insert OLD version into document_version first, then UPDATE document
  db.transaction((tx) => {
    tx.insert(documentVersion).values({
      id: crypto.randomUUID(),
      documentId: doc.id,
      versionNumber: doc.version ?? 1,
      storagePath: doc.storagePath,
      sha256: doc.sha256,
      size: doc.size,
      uploadedAt: doc.uploadedAt,
    }).run();
    tx.update(document).set({
      storagePath: newPath,
      sha256,
      size: buffer.byteLength,
      version: newVersionNumber,
    }).where(eq(document.id, documentId)).run();
  });

  return { ok: true, newVersion: newVersionNumber };
}
```
