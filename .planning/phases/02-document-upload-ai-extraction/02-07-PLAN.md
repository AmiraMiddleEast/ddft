Create `__tests__/phase2-integration.test.ts` (extend the Phase 1 integration pattern). Use `// @vitest-environment node`. Mock `next/headers` and `@/lib/auth` at the top. Mock `@/lib/extraction/claude` (the SDK wrapper — NEVER hit the real API).

```ts
// @vitest-environment node
import { describe, it, expect, beforeEach, vi } from "vitest";
import { readFile } from "node:fs/promises";
import path from "node:path";

vi.mock("next/headers", () => ({ headers: async () => new Headers() }));

const sessionHolder = { current: null as null | { user: { id: string; email: string }; session: { id: string } } };

vi.mock("@/lib/auth", () => ({
  auth: {
    api: {
      getSession: vi.fn(async () => sessionHolder.current),
    },
  },
}));

vi.mock("@/lib/extraction/claude", () => ({
  extractFields: vi.fn(),
}));

import { db } from "@/db/client";
import { document, extraction, extractionLog, user } from "@/db/schema";
import { uploadSingleDocumentAction } from "@/lib/uploads/actions";
import { extractDocumentAction } from "@/lib/extraction/actions";
import { extractFields } from "@/lib/extraction/claude";
import { GET as getPdf } from "@/app/api/documents/[id]/pdf/route";

const OK = {
  parsed: {
    dokumenten_typ: { value: "Zeugnis", confidence: "high" as const, reasoning: "title" },
    ausstellende_behoerde: { value: "Universität Heidelberg", confidence: "high" as const, reasoning: "seal" },
    ausstellungsort: { value: "Heidelberg", confidence: "high" as const, reasoning: "header" },
    bundesland: { value: "Baden-Württemberg", confidence: "medium" as const, reasoning: "inferred" },
    ausstellungsdatum: { value: "2021-06-30", confidence: "high" as const, reasoning: "stamped" },
    voller_name: { value: "Max Mustermann", confidence: "high" as const, reasoning: "name line" },
  },
  usage: { input_tokens: 5000, output_tokens: 400 },
  model: "claude-sonnet-4-20250514",
};

async function fd(file: File) {
  const f = new FormData();
  f.append("file", file);
  return f;
}

beforeEach(async () => {
  await db.delete(extractionLog);
  await db.delete(extraction);
  await db.delete(document);
  await db.delete(user);
  await db.insert(user).values([
    { id: "u-a", name: "Alice", email: "a@x.de", emailVerified: true },
    { id: "u-b", name: "Bob", email: "b@x.de", emailVerified: true },
  ]);
  sessionHolder.current = { user: { id: "u-a", email: "a@x.de" }, session: { id: "s-a" } };
  vi.mocked(extractFields).mockReset();
});

describe("Phase 2 — integration", () => {
  it("uploads, extracts, and renders data for the owning user", async () => {
    const bytes = await readFile(path.resolve(process.cwd(), "transcript.pdf"));
    const up = await uploadSingleDocumentAction(
      null,
      await fd(new File([bytes], "transcript.pdf", { type: "application/pdf" })),
    );
    expect(up.ok).toBe(true);
    if (!up.ok) return;
    expect(up.dedup).toBe(false);

    vi.mocked(extractFields).mockResolvedValueOnce(OK as any);
    const ex = await extractDocumentAction(up.documentId);
    expect(ex.ok).toBe(true);

    const docs = await db.select().from(document);
    expect(docs.length).toBe(1);
    expect(docs[0].extractionStatus).toBe("done");

    const xs = await db.select().from(extraction);
    expect(xs.length).toBe(6);
    const logs = await db.select().from(extractionLog);
    expect(logs.length).toBe(1);
    expect(logs[0].inputTokens).toBe(5000);
    expect(logs[0].outputTokens).toBe(400);
    expect(logs[0].costEur).toBeGreaterThan(0);
  });

  it("dedups the same file on second upload (no new document row)", async () => {
    const bytes = await readFile(path.resolve(process.cwd(), "transcript.pdf"));
    const r1 = await uploadSingleDocumentAction(
      null,
      await fd(new File([bytes], "one.pdf", { type: "application/pdf" })),
    );
    const r2 = await uploadSingleDocumentAction(
      null,
      await fd(new File([bytes], "two.pdf", { type: "application/pdf" })),
    );
    expect(r1.ok && r2.ok).toBe(true);
    if (r1.ok && r2.ok) {
      expect(r2.documentId).toBe(r1.documentId);
      expect(r2.dedup).toBe(true);
    }
    const docs = await db.select().from(document);
    expect(docs.length).toBe(1);
  });

  it("returns 404 when a user requests another user's PDF via the Route Handler", async () => {
    const bytes = await readFile(path.resolve(process.cwd(), "transcript.pdf"));
    const up = await uploadSingleDocumentAction(
      null,
      await fd(new File([bytes], "a.pdf", { type: "application/pdf" })),
    );
    expect(up.ok).toBe(true);
    if (!up.ok) return;

    // Switch to user B.
    sessionHolder.current = { user: { id: "u-b", email: "b@x.de" }, session: { id: "s-b" } };
    const res = await getPdf(new Request("http://localhost/api/documents/x/pdf"), {
      params: Promise.resolve({ id: up.documentId }),
    } as any);
    expect(res.status).toBe(404);
  });

  it("returns 401 when unauthenticated", async () => {
    sessionHolder.current = null;
    const res = await getPdf(new Request("http://localhost/api/documents/x/pdf"), {
      params: Promise.resolve({ id: "any" }),
    } as any);
    expect(res.status).toBe(401);
  });
});
```

Also create `scripts/seed-extraction-fixture.mjs` — a one-shot Node ESM script for the human verification step. It seeds a user, a document, 6 extraction rows, and one log row, pointing the document's storagePath at `transcript.pdf` at the repo root (read-only — never moved). Usage: `node scripts/seed-extraction-fixture.mjs`.

```js
// scripts/seed-extraction-fixture.mjs
import { db } from "../db/client.ts";
import { document, extraction, extractionLog, user } from "../db/schema.ts";
import path from "node:path";
import { writeFile, mkdir } from "node:fs/promises";
import { readFile } from "node:fs/promises";

const USER_ID = "seed-user";
const DOC_ID = "seed-doc-1";

await db.insert(user).values({
  id: USER_ID, name: "Seed User", email: "seed@angela.local", emailVerified: true,
}).onConflictDoNothing();

const transcript = await readFile(path.resolve(process.cwd(), "transcript.pdf"));
await mkdir(path.resolve(process.cwd(), "data/uploads"), { recursive: true });
await writeFile(path.resolve(process.cwd(), `data/uploads/${DOC_ID}.pdf`), transcript);

await db.insert(document).values({
  id: DOC_ID, userId: USER_ID, filename: "transcript.pdf", size: transcript.byteLength,
  sha256: "seed-fixture-hash", storagePath: `data/uploads/${DOC_ID}.pdf`,
  extractionStatus: "done", extractedAt: new Date(),
}).onConflictDoNothing();

const FIELDS = [
  ["dokumenten_typ",        "Zeugnis",              "high"],
  ["ausstellende_behoerde", "Universität Heidelberg","high"],
  ["ausstellungsort",       "Heidelberg",           "high"],
  ["bundesland",            "Baden-Württemberg",    "medium"],
  ["ausstellungsdatum",     "2021-06-30",           "high"],
  ["voller_name",           "Max Mustermann",       "high"],
];
for (const [name, value, conf] of FIELDS) {
  await db.insert(extraction).values({
    id: `seed-ext-${name}`, documentId: DOC_ID, fieldName: name,
    fieldValue: value, confidence: conf, reasoning: "seeded",
  }).onConflictDoNothing();
}

await db.insert(extractionLog).values({
  id: "seed-log-1", documentId: DOC_ID,
  inputTokens: 5000, outputTokens: 400, costEur: 0.025, claudeModel: "claude-sonnet-4-20250514",
}).onConflictDoNothing();

console.log(`Seeded: user=${USER_ID} document=${DOC_ID}`);
console.log(`Visit /documents/${DOC_ID} after signing in as seed@angela.local`);
```
