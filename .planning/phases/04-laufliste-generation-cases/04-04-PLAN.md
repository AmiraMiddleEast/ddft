Implementation skeleton (from RESEARCH Code Examples — adapt):
```typescript
"use server";
import { headers } from "next/headers";
import { revalidatePath } from "next/cache";
import { auth } from "@/lib/auth";
import { db as defaultDb } from "@/db/client";
import { caseTable, laufliste } from "@/db/schema";
import { buildLauflisteInput } from "./build-input";
import { renderLaufliste } from "./pdf/render";
import { writeLauflisteToDisk } from "./storage";
import { eq } from "drizzle-orm";

export async function generateLauflisteAction(caseId: string, opts?: { db?: DrizzleDb; session?: Session }) {
  const session = opts?.session ?? await auth.api.getSession({ headers: await headers() });
  if (!session) return { ok: false as const, error: "UNAUTHORIZED" as const };
  const db = opts?.db ?? defaultDb;

  const input = await buildLauflisteInput(caseId, session.user.id, db);
  if (!input) return { ok: false as const, error: "NOT_FOUND" as const };
  if (input.documents.length === 0) return { ok: false as const, error: "EMPTY_CASE" as const };
  // review_status check: buildLauflisteInput joins documentReview; executor augments input
  // OR re-check here via a direct query. Pick: augment input with perDocReviewStatus OR
  // pass documents through a separate validator. RECOMMENDED: add a dedicated
  // precondition query `countUnapprovedInCase(caseId, userId, db)` called before render.

  let bytes: Buffer;
  try {
    bytes = await renderLaufliste(input);
  } catch (e) {
    return { ok: false as const, error: "RENDER_FAILED" as const, details: String(e) };
  }

  const storagePath = await writeLauflisteToDisk(caseId, bytes);
  const id = crypto.randomUUID();

  // better-sqlite3 sync transaction — STATE.md Phase 2 Plan 04 precedent
  db.transaction((tx) => {
    tx.insert(laufliste).values({
      id, caseId, userId: session.user.id,
      pdfStoragePath: storagePath,
      documentCount: input.documents.length,
      fileSize: bytes.byteLength,
    }).run();
    tx.update(caseTable)
      .set({ status: "pdf_generated" })
      .where(eq(caseTable.id, caseId))
      .run();
  });

  revalidatePath(`/cases/${caseId}`, "page");
  return { ok: true as const, lauflisteId: id };
}
```

Note on D-17 validation: since buildLauflisteInput doesn't currently return review_status, add a precondition query `countUnapprovedDocsInCase(caseId, db)` to lib/laufliste/queries.ts, called right after the empty check. Return UNREVIEWED_DOCS if > 0.

Note on render failure: If renderToBuffer throws, we must NOT have a dangling file on disk (we haven't written yet — good). Use the ordering above (render FIRST, write SECOND, insert THIRD) so failures cascade cleanly without cleanup.
