import { z } from "zod";

export const CorrectedFieldsSchema = z.object({
  dokumenten_typ: z.string().max(200),
  ausstellende_behoerde: z.string().max(300),
  ausstellungsort: z.string().max(200),
  bundesland: z.string().max(100),
  ausstellungsdatum: z
    .string()
    .regex(/^\d{4}-\d{2}-\d{2}$/, { message: "Bitte ein gültiges Datum eingeben." })
    .or(z.literal("")),
  voller_name: z.string().max(300),
});

export type CorrectedFields = z.infer<typeof CorrectedFieldsSchema>;

export const ApproveSchema = z.object({
  documentId: z.string().min(1),
  corrected: CorrectedFieldsSchema,
});

export type ApproveInput = z.infer<typeof ApproveSchema>;

// Used by chooseAmbiguousAuthority action
export const ChooseAuthoritySchema = z.object({
  documentId: z.string().min(1),
  authorityId: z.string().min(1),
});
export type ChooseAuthorityInput = z.infer<typeof ChooseAuthoritySchema>;

```ts
"use server";

import { headers } from "next/headers";
import { and, eq } from "drizzle-orm";
import { randomUUID } from "node:crypto";
import { z } from "zod";

import { auth } from "@/lib/auth";
import { db } from "@/db";
import { document, documentReview, behoerdenAuthority } from "@/db/schema";
import { ApproveSchema, ChooseAuthoritySchema } from "@/lib/validations/review";
import { resolveAuthority } from "@/lib/behoerden/resolve";

type Ok<T> = { ok: true; data: T };
type Err = { ok: false; error: string };

// ---- approveAndResolve ----

export async function approveAndResolve(input: {
  documentId: string;
  corrected: {
    dokumenten_typ: string;
    ausstellende_behoerde: string;
    ausstellungsort: string;
    bundesland: string;
    ausstellungsdatum: string;
    voller_name: string;
  };
}): Promise<Ok<Awaited<ReturnType<typeof resolveAuthority>>> | Err> {
  const session = await auth.api.getSession({ headers: await headers() });
  if (!session) return { ok: false, error: "unauthorized" };

  const parsed = ApproveSchema.safeParse(input);
  if (!parsed.success) return { ok: false, error: "invalid_input" };

  // Ownership
  const docRows = await db
    .select()
    .from(document)
    .where(and(eq(document.id, parsed.data.documentId), eq(document.userId, session.user.id)));
  const doc = docRows[0];
  if (!doc) return { ok: false, error: "not_found" };

  // Resolve
  const result = await resolveAuthority(parsed.data.corrected, db);
  const resolvedAuthorityId =
    result.status === "matched" ? result.authority.id : null;
  const lookupStatus = result.status; // enum literal — 'matched' | 'ambiguous' | 'not_found'

  // Transactional persist (upsert logic: single row per document)
  db.transaction((tx) => {
    const existing = tx
      .select()
      .from(documentReview)
      .where(eq(documentReview.documentId, parsed.data.documentId))
      .all();

    if (existing.length > 0) {
      tx.update(documentReview)
        .set({
          approvedByUserId: session.user.id,
          approvedAt: new Date(),
          correctedFields: parsed.data.corrected,
          resolvedAuthorityId,
          lookupStatus,
        })
        .where(eq(documentReview.id, existing[0].id))
        .run();
    } else {
      tx.insert(documentReview)
        .values({
          id: randomUUID(),
          documentId: parsed.data.documentId,
          approvedByUserId: session.user.id,
          correctedFields: parsed.data.corrected,
          resolvedAuthorityId,
          lookupStatus,
        })
        .run();
    }

    tx.update(document)
      .set({ reviewStatus: "approved", reviewedAt: new Date() })
      .where(eq(document.id, parsed.data.documentId))
      .run();
  });

  return { ok: true, data: result };
}

// ---- chooseAmbiguousAuthority ----

export async function chooseAmbiguousAuthority(input: {
  documentId: string;
  authorityId: string;
}): Promise<Ok<{ authority: typeof behoerdenAuthority.$inferSelect }> | Err> {
  const session = await auth.api.getSession({ headers: await headers() });
  if (!session) return { ok: false, error: "unauthorized" };

  const parsed = ChooseAuthoritySchema.safeParse(input);
  if (!parsed.success) return { ok: false, error: "invalid_input" };

  const docRows = await db
    .select()
    .from(document)
    .where(and(eq(document.id, parsed.data.documentId), eq(document.userId, session.user.id)));
  if (!docRows[0]) return { ok: false, error: "not_found" };

  const reviewRows = await db
    .select()
    .from(documentReview)
    .where(eq(documentReview.documentId, parsed.data.documentId));
  const review = reviewRows[0];
  if (!review || review.lookupStatus !== "ambiguous") {
    return { ok: false, error: "invalid_choice" };
  }

  const authRows = await db
    .select()
    .from(behoerdenAuthority)
    .where(eq(behoerdenAuthority.id, parsed.data.authorityId));
  const chosen = authRows[0];
  if (!chosen) return { ok: false, error: "invalid_choice" };

  db.transaction((tx) => {
    tx.update(documentReview)
      .set({
        resolvedAuthorityId: chosen.id,
        lookupStatus: "matched",
        approvedAt: new Date(),
      })
      .where(eq(documentReview.id, review.id))
      .run();
  });

  return { ok: true, data: { authority: chosen } };
}
```

|----------|-------------|
| Client → Server Action | User-submitted corrected fields cross network to Node runtime |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-04-01 | Elevation of Privilege | User approves a document owned by another user | mitigate | Ownership predicate `eq(document.userId, session.user.id)` BEFORE any mutation (Test 2) |
| T-03-04-02 | Spoofing | Unauthenticated caller invokes Server Action directly | mitigate | `auth.api.getSession` returns null → action returns `unauthorized` before any DB read (Test 1) |
| T-03-04-03 | Input Validation / Injection | Over-long fields cause DoS or DB strain | mitigate | Zod `.max()` caps per field; action returns `invalid_input` before reaching resolver or DB (Test 3) |
| T-03-04-04 | Tampering | Client POSTs `authorityId` that does not belong to the resolved (state, doc_type) set | mitigate | `chooseAmbiguousAuthority` verifies review.lookupStatus === 'ambiguous' AND authority exists; rejects otherwise (Test 10) |
| T-03-04-05 | Repudiation | No audit trail of who approved what | mitigate | documentReview.approvedByUserId + approvedAt columns record the actor + timestamp (Plan 01 schema) |
| T-03-04-06 | Information Disclosure | Error messages leak internal row data | mitigate | Err variant returns short stable codes (`unauthorized`, `not_found`, `invalid_input`, `invalid_choice`) — no row details |
| T-03-04-07 | DoS | Many rapid approvals storm DB | accept | Single-user internal tool (per CLAUDE.md constraints); SQLite WAL handles bursty single-writer load |
</threat_model>
