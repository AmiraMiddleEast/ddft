Create `scripts/seed-behoerden.ts`. The script has three phases: LOAD RAW → PARSE (cached) → INSERT (idempotent).

```ts
// scripts/seed-behoerden.ts
import { db } from "../db";
import {
  behoerdenState, behoerdenRegierungsbezirk,
  behoerdenDocumentType, behoerdenAuthority,
} from "../db/schema";
import { parseStateWithClaude, StateParseOutput, normalizeDocTypeSlug } from "./parse-state-with-claude";
import fs from "node:fs";
import path from "node:path";
import { z } from "zod";
// id generator — crypto.randomUUID is fine
import { randomUUID } from "node:crypto";

const CACHE_PATH = path.resolve(process.cwd(), "data/behoerden-parsed.json");
const SOURCE_PATH = path.resolve(process.cwd(), "behoerden_db.json");

const RawStateSchema = z.object({
  bundesland: z.string(),
  hat_regierungsbezirke: z.boolean().default(false),
  besonderheiten: z.string().nullable().optional(),
  dokumente_raw: z.string(),
});

const RawFileSchema = z.array(RawStateSchema);

const CacheSchema = z.array(StateParseOutput);

export async function seedBehoerden(opts: { force?: boolean } = {}) {
  // 1. LOAD RAW
  const rawJson = JSON.parse(fs.readFileSync(SOURCE_PATH, "utf8"));
  const rawStates = RawFileSchema.parse(rawJson);
  console.log(`[seed-behoerden] loaded ${rawStates.length} states from ${SOURCE_PATH}`);

  // 2. PARSE (cached unless --force)
  let parsed: z.infer<typeof CacheSchema>;
  if (fs.existsSync(CACHE_PATH) && !opts.force) {
    console.log(`[seed-behoerden] cache hit: ${CACHE_PATH} — skipping Claude parse`);
    parsed = CacheSchema.parse(JSON.parse(fs.readFileSync(CACHE_PATH, "utf8")));
  } else {
    console.log(`[seed-behoerden] parsing 16 states via Claude (this costs ~$0.30 + a few minutes)`);
    const results: z.infer<typeof CacheSchema> = [];
    for (const raw of rawStates) {
      const slugHint = normalizeDocTypeSlug(raw.bundesland);
      console.log(`  → parsing ${raw.bundesland} (${raw.dokumente_raw.length} chars)…`);
      const result = await parseStateWithClaude(raw.dokumente_raw, slugHint);
      results.push(result);
    }
    fs.mkdirSync(path.dirname(CACHE_PATH), { recursive: true });
    fs.writeFileSync(CACHE_PATH, JSON.stringify(results, null, 2));
    parsed = results;
    console.log(`[seed-behoerden] wrote cache: ${CACHE_PATH}`);
  }

  // 3. INSERT (idempotent: wipe + re-insert in a transaction; safer than attempting diff-merge)
  db.transaction((tx) => {
    // Delete in FK-safe order.
    tx.delete(behoerdenAuthority).run();
    tx.delete(behoerdenRegierungsbezirk).run();
    tx.delete(behoerdenDocumentType).run();
    tx.delete(behoerdenState).run();

    // GLOBAL document_type dedup — aggregate slugs across all states first.
    const docTypeMap = new Map<string, { id: string; displayName: string }>();
    for (const state of parsed) {
      for (const a of state.authorities) {
        if (!docTypeMap.has(a.document_type_slug)) {
          docTypeMap.set(a.document_type_slug, {
            id: a.document_type_slug,
            displayName: a.document_type_display,
          });
        }
      }
    }
    for (const dt of docTypeMap.values()) {
      tx.insert(behoerdenDocumentType).values(dt).run();
    }

    // States + regierungsbezirke + authorities
    for (const state of parsed) {
      tx.insert(behoerdenState).values({
        id: state.state_slug,
        name: state.state_name,
        hatRegierungsbezirke: state.hat_regierungsbezirke,
        besonderheiten: state.besonderheiten ?? null,
      }).run();

      const rbzIdByName = new Map<string, string>();
      for (const rbzName of state.regierungsbezirke ?? []) {
        const rbzSlug = normalizeDocTypeSlug(rbzName);
        const id = `${state.state_slug}-${rbzSlug}`;
        tx.insert(behoerdenRegierungsbezirk).values({
          id,
          stateId: state.state_slug,
          name: rbzName,
          slug: rbzSlug,
        }).run();
        rbzIdByName.set(rbzName, id);
      }

      for (const a of state.authorities) {
        const rbzId = a.regierungsbezirk ? rbzIdByName.get(a.regierungsbezirk) ?? null : null;
        tx.insert(behoerdenAuthority).values({
          id: randomUUID(),
          stateId: state.state_slug,
          regierungsbezirkId: rbzId,
          documentTypeId: a.document_type_slug,
          name: a.name,
          address: a.address,
          phone: a.phone,
          email: a.email,
          website: a.website,
          officeHours: a.office_hours,
          notes: a.notes,
          specialRules: a.special_rules,
          needsReview: a.needs_review,
        }).run();
      }
    }
  });

  // Sanity guard (Pitfall 5): count needs_review=true authorities.
  const review = db.select().from(behoerdenAuthority).where(/* needsReview=1 */ undefined as any).all();
  // (Use a proper drizzle eq() predicate here; shown schematically.)
  const reviewCount = review.filter((r: any) => r.needsReview).length;
  console.log(`[seed-behoerden] inserted: states=${parsed.length}, authorities total, needs_review=${reviewCount}`);
  if (reviewCount < 20) {
    console.warn(`[seed-behoerden] WARNING: needs_review count (${reviewCount}) is below the expected ≥24 baseline from research Pitfall 5 — parse may have dropped [PRÜFEN] markers.`);
  }

  return { states: parsed.length, needsReviewCount: reviewCount };
}

// CLI entry
if (import.meta.url === `file://${process.argv[1]}`) {
  const force = process.argv.includes("--force");
  seedBehoerden({ force }).then(() => process.exit(0)).catch((e) => {
    console.error(e);
    process.exit(1);
  });
}
```

| Threat ID | Category | Component | Disposition | Mitigation Plan |

| T-03-02-02 | Info Disclosure | Claude request contains German authority markdown | accept | Content is public-ish state-authored data; no user PII or secrets. Documented in research §Security "Seed script exfiltrates DB to Claude" |
