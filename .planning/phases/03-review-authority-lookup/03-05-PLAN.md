Step 1 — Create `app/(app)/documents/[id]/review/page.tsx` as an async Server Component:

```tsx
import { redirect, notFound } from "next/navigation";
import Link from "next/link";
import { headers } from "next/headers";
import { and, eq } from "drizzle-orm";

import { auth } from "@/lib/auth";
import { db } from "@/db";
import { document, extraction } from "@/db/schema";
import { listDocumentTypes, listStates } from "@/lib/behoerden/queries";
import { PdfPreview } from "../_components/PdfPreview"; // Phase 2
import { ReviewForm } from "./_components/ReviewForm";
import { Card } from "@/components/ui/card";

type Props = { params: Promise<{ id: string }> };

export const metadata = { title: "Überprüfung — Angela" };

export default async function ReviewPage({ params }: Props) {
  const { id } = await params;

  const session = await auth.api.getSession({ headers: await headers() });
  if (!session) redirect("/sign-in");

  const docRows = await db
    .select()
    .from(document)
    .where(and(eq(document.id, id), eq(document.userId, session.user.id)));
  const doc = docRows[0];
  if (!doc) notFound();

  // Only 'done' status may review; otherwise kick back to detail.
  if (doc.extractionStatus !== "done") {
    redirect(`/documents/${id}`);
  }

  const extractions = await db.select().from(extraction).where(eq(extraction.documentId, id));

  // Map extractions into { fieldName: { value, confidence } }
  const original: Record<string, { value: string; confidence: "high" | "medium" | "low" }> = {};
  for (const e of extractions) {
    original[e.fieldName] = {
      value: e.fieldValue ?? "",
      confidence: e.confidence,
    };
  }

  // Parallel data fetch for Select dropdowns
  const [docTypes, states] = await Promise.all([listDocumentTypes(), listStates()]);

  return (
    <main className="mx-auto w-full max-w-6xl px-6 py-8">
      {/* Breadcrumb */}
      <nav className="mb-4 text-sm text-muted-foreground">
        <Link href="/" className="hover:underline">Übersicht</Link>
        {" / "}
        <Link href={`/documents/${id}`} className="hover:underline">Dokument</Link>
        {" / "}
        <span aria-current="page">Überprüfung</span>
      </nav>

      <header className="mb-8">
        <h1 className="text-2xl font-semibold">Überprüfung</h1>
        <p className="mt-2 text-sm text-muted-foreground">
          {doc.filename} · Hochgeladen am{" "}
          {new Intl.DateTimeFormat("de-DE", {
            day: "2-digit", month: "2-digit", year: "numeric",
            hour: "2-digit", minute: "2-digit",
          }).format(doc.uploadedAt as Date)}
        </p>
      </header>

      <div className="grid gap-6 lg:grid-cols-[1fr_480px] lg:gap-8">
        <Card className="p-0 overflow-hidden">
          <div className="border-b px-6 py-3 text-sm font-semibold">Originaldokument</div>
          <PdfPreview documentId={id} className="min-h-[640px]" />
        </Card>

        <Card className="p-6">
          <ReviewForm
            documentId={id}
            original={original}
            documentTypes={docTypes}
            states={states}
          />
        </Card>
      </div>
    </main>
  );
}
```

Step 3 — ReviewForm component (`_components/ReviewForm.tsx`), the main client orchestrator:

```tsx
"use client";

import { useState, useTransition } from "react";
import { toast } from "sonner";
import { Select, SelectTrigger, SelectContent, SelectItem, SelectValue } from "@/components/ui/select";
import { Input } from "@/components/ui/input";
import { Button } from "@/components/ui/button";
import { Label } from "@/components/ui/label";
import { FieldRow } from "./FieldRow";
import { DiscardDialog } from "./DiscardDialog";
import { AuthorityResultPanel } from "./AuthorityResultPanel";
import { CorrectedFieldsSchema, type CorrectedFields } from "@/lib/validations/review";
import { approveAndResolve, chooseAmbiguousAuthority } from "@/lib/review/actions";

const FIELD_NAMES = [
  "dokumenten_typ",
  "ausstellende_behoerde",
  "ausstellungsort",
  "bundesland",
  "ausstellungsdatum",
  "voller_name",
] as const;

type Original = Record<string, { value: string; confidence: "high" | "medium" | "low" }>;

type Props = {
  documentId: string;
  original: Original;
  documentTypes: { id: string; displayName: string }[];
  states: { id: string; name: string }[];
};

export function ReviewForm({ documentId, original, documentTypes, states }: Props) {
  const [values, setValues] = useState<CorrectedFields>(() => ({
    dokumenten_typ: original.dokumenten_typ?.value ?? "",
    ausstellende_behoerde: original.ausstellende_behoerde?.value ?? "",
    ausstellungsort: original.ausstellungsort?.value ?? "",
    bundesland: original.bundesland?.value ?? "",
    ausstellungsdatum: original.ausstellungsdatum?.value ?? "",
    voller_name: original.voller_name?.value ?? "",
  }));
  const [errors, setErrors] = useState<Partial<Record<keyof CorrectedFields, string>>>({});
  const [pending, start] = useTransition();
  const [result, setResult] = useState<any>(null);          // ResolverResult after submit
  const [confirmDiscard, setConfirmDiscard] = useState(false);

  const isDirty = (k: keyof CorrectedFields) => values[k] !== (original[k]?.value ?? "");
  const anyDirty = FIELD_NAMES.some((k) => isDirty(k));

  function setField<K extends keyof CorrectedFields>(k: K, v: CorrectedFields[K]) {
    setValues((prev) => ({ ...prev, [k]: v }));
    if (errors[k]) setErrors((prev) => ({ ...prev, [k]: undefined }));
  }

  async function onSubmit(e: React.FormEvent) {
    e.preventDefault();
    const parsed = CorrectedFieldsSchema.safeParse(values);
    if (!parsed.success) {
      const next: typeof errors = {};
      for (const issue of parsed.error.issues) {
        const key = issue.path[0] as keyof CorrectedFields;
        next[key] = issue.message;
      }
      setErrors(next);
      // Focus first invalid field
      const firstKey = FIELD_NAMES.find((k) => next[k]);
      if (firstKey) document.getElementById(`field-${firstKey}`)?.focus();
      return;
    }
    start(async () => {
      try {
        const res = await approveAndResolve({ documentId, corrected: parsed.data });
        if (!res.ok) {
          toast.error("Behörde konnte nicht ermittelt werden. Bitte erneut versuchen.");
          return;
        }
        setResult(res.data);
        if (res.data.status === "matched") toast.success("Zuständige Behörde ermittelt.");
        // Scroll to result panel
        setTimeout(() => {
          document.getElementById("authority-result")?.scrollIntoView({ behavior: "smooth" });
        }, 50);
      } catch {
        toast.error("Behörde konnte nicht ermittelt werden. Bitte erneut versuchen.");
      }
    });
  }

  function onDiscard() {
    setConfirmDiscard(false);
    setValues({
      dokumenten_typ: original.dokumenten_typ?.value ?? "",
      ausstellende_behoerde: original.ausstellende_behoerde?.value ?? "",
      ausstellungsort: original.ausstellungsort?.value ?? "",
      bundesland: original.bundesland?.value ?? "",
      ausstellungsdatum: original.ausstellungsdatum?.value ?? "",
      voller_name: original.voller_name?.value ?? "",
    });
    setErrors({});
    setResult(null);
  }

  async function onChooseAmbiguous(authorityId: string) {
    const res = await chooseAmbiguousAuthority({ documentId, authorityId });
    if (!res.ok) {
      toast.error("Behörde konnte nicht ermittelt werden. Bitte erneut versuchen.");
      return;
    }
    setResult({ status: "matched", authority: res.data.authority, routing_path: [], special_rules: res.data.authority.specialRules, needs_review: res.data.authority.needsReview });
    toast.success("Zuständige Behörde ermittelt.");
  }

  return (
    <>
      <h2 className="mb-4 text-2xl font-semibold">Extrahierte Daten</h2>
      <form onSubmit={onSubmit} className="grid gap-4">
        {/* Render a FieldRow per field. Examples: */}

        <FieldRow
          label="Dokumenttyp"
          name="dokumenten_typ"
          value={values.dokumenten_typ}
          originalValue={original.dokumenten_typ?.value ?? ""}
          confidence={original.dokumenten_typ?.confidence ?? "low"}
          isDirty={isDirty("dokumenten_typ")}
          error={errors.dokumenten_typ}
        >
          <Select value={values.dokumenten_typ} onValueChange={(v) => setField("dokumenten_typ", v)}>
            <SelectTrigger id="field-dokumenten_typ">
              <SelectValue placeholder="Dokumenttyp auswählen" />
            </SelectTrigger>
            <SelectContent>
              {documentTypes.map((dt) => (
                <SelectItem key={dt.id} value={dt.displayName}>{dt.displayName}</SelectItem>
              ))}
            </SelectContent>
          </Select>
        </FieldRow>

        {/* Ausstellende Behörde — Input, label "Ausstellende Behörde" */}
        {/* Ausstellungsort — Input, label "Ausstellungsort" */}
        {/* Bundesland — Select (states + "Unbekannt / Sonstiges" sentinel) */}
        {/* Ausstellungsdatum — Input type="date" */}
        {/* Voller Name — Input */}

        <div className="mt-4 flex items-center justify-between gap-4">
          <Button
            type="button"
            variant="ghost"
            disabled={!anyDirty}
            onClick={() => setConfirmDiscard(true)}
          >
            Verwerfen
          </Button>
          <Button type="submit" disabled={pending}>
            {pending ? "Behörde wird ermittelt …" : "Speichern & Behörde ermitteln"}
          </Button>
        </div>
      </form>

      {result && (
        <div id="authority-result" className="mt-8">
          <AuthorityResultPanel
            result={result}
            onChooseAuthority={onChooseAmbiguous}
            onAdjustInputs={() => {
              setResult(null);
              document.getElementById("field-dokumenten_typ")?.focus();
            }}
          />
        </div>
      )}

      <DiscardDialog
        open={confirmDiscard}
        onConfirm={onDiscard}
        onCancel={() => setConfirmDiscard(false)}
      />
    </>
  );
}
```
